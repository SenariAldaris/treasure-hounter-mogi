Предложу свой вариант, переборщик Came. Испытывал на парковочном барьере.
Код HTML:
int value = 20;       // переменная для хранения значения ШИМ
int ShimPin = 9;      // светодиод подключён к выводу 9

#include <LiquidCrystal.h> // include the library code:
LiquidCrystal lcd(12, 11, 6, 5, 4, 3);

//Возможное количество кодовых комбинаций – 4096.
#define SvetPin 13
#define txPin 8
int var=0;
void setup(){
  lcd.begin(16, 2);
  lcd.print("Send came...");

  Serial.begin(115200); 
  pinMode(txPin, OUTPUT);
  pinMode(SvetPin, OUTPUT);
}

void loop()
{
  analogWrite(ShimPin, value); //Шим

  SendCame(var);
  var=var+1;
  //  Serial.println(var);     
  lcd.setCursor(5, 1);
  lcd.print(var);
  if (var > 4096) 
  {
    lcd.clear();
    lcd.begin(16, 2);
    lcd.print("Pauza");
    var=0; 
    digitalWrite(SvetPin, HIGH);
    delay(5000);  
    digitalWrite(SvetPin, LOW);
    lcd.begin(16, 2);
    lcd.print("Send came...");  
  }
}

void SendCame(long Code)
{     
  for (int i=0; i<4; i++) // посылку посылаем 4 раза подряд.
  {
    Send_preambula();
    for (int i=12; i>0; i--){
      SendBit(bitRead(Code,i-1)); // побитово перебираем и посылаем код
    }
    digitalWrite(txPin, LOW); 
    delayMicroseconds(11520);
  }
}

void SendBit(byte b){
  if (b) {
    digitalWrite(txPin, LOW);   // 1
    delayMicroseconds(640);
    digitalWrite(txPin, HIGH);
    delayMicroseconds(320);
  }
  else {  
    digitalWrite(txPin, LOW);    // 0
    delayMicroseconds(320);
    digitalWrite(txPin, HIGH);
    delayMicroseconds(640);
  }
}

void Send_preambula()
{
  digitalWrite(txPin, HIGH);   
  delayMicroseconds(320);
  digitalWrite(txPin, LOW);
}
